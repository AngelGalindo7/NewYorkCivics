# NYC Civic Data Ingestion System — RUNTIME dependencies.
# Dep manager is pip + venv (project choice). pyproject.toml holds TOOL CONFIG ONLY,
# not dependency lists. Dev/eval deps live in requirements-dev.txt.
#
# Pins are loose (>=) on purpose: this is a greenfield scaffold and we want the latest
# compatible patch/minor. Tighten to == once a known-good lockfile exists (Phase 1+).
#
# PYTHON VERSION CAVEAT: target is Python 3.14 (see README setup), but on
# 3.14 some binary wheels still lag (notably psycopg, PyMuPDF, and geosupport). If `pip install` fails
# to find a wheel for your platform, fall back to Python 3.11–3.12 — the safe baseline.

# --- PDF parsing (Stage: Parse — digital text, fast + free) ---
pdfplumber>=0.11        # primary digital-PDF text + layout extraction
PyMuPDF>=1.24           # page rasterization for the vision fallback; faster on big PDFs

# --- Schemas / validation (Stage: Extract — clean record shape, Rule 3 provenance) ---
pydantic>=2.7           # canonical event + Provenance models

# --- Structured source connectors (Stage: Fetch — NYC-SPECIFIC, no LLM per Rule 1) ---
sodapy>=2.2                     # Socrata / SODA client (DOB NOW, HPD violations)
# NYC Council / Legistar ingestion uses DataMade's python-legistar-scraper, which is NOT
# published on PyPI — so it is intentionally NOT listed here (a bare pip install of it fails
# with "No matching distribution found", which would break `pip install -r requirements.txt`).
# Install it from its Git repo when the Legistar connector lands (Phase 1), pinning a commit:
#   pip install "python-legistar @ git+https://github.com/opencivicdata/python-legistar-scraper.git@<commit>"
# TODO Phase 1: confirm the canonical repo URL + pin a commit, then document in README setup.

# --- Storage (Stage: Store — Postgres + PostGIS) ---
psycopg[binary]>=3.1    # Postgres driver (binary extra ships the libpq wheel)
SQLAlchemy>=2.0         # connection / core; canonical schema lives in store/schema.sql

# --- HTTP + config ---
httpx>=0.27             # async-capable HTTP client for fetchers
python-dotenv>=1.0      # load .env (config-driven flags, Rule 6) — never commit .env

# --- LLMs (Stage: Extract + Eval judge — model name behind config flag, Rule 6) ---
google-genai>=1.0       # DEFAULT extractor: EXTRACT_MODEL=gemini-2.5-flash (cheapest vision; new `from google import genai` SDK)
anthropic>=0.34         # cross-family judge: JUDGE_MODEL=claude-haiku-4-5 (Rule 12)
openai>=1.40            # OPTIONAL alternate judge (e.g. gpt-4o-mini); unset by default

# --- Reliability + geometry ---
tenacity>=8.2           # retry/backoff for flaky source endpoints (Rule 2: fail fast, then quarantine)
shapely>=2.0            # geometry helpers feeding PostGIS point/radius matching
